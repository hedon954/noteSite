# Mac 上如何通过进程号查看其占用的端口

 lsof -nP -p <进程号> | grep LISTEN

ps -ef | grep <PNAME> | awk 'NR==1{print $2}'

lsof -nP -p <PID> | grep LISTEN | awk '{print $9}'
