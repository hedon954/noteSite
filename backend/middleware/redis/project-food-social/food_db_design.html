# t_diner

字段名 数据类型 说明
id int 主键
username varchar(255)
phone varchar(255)
email varchar(255)
password varchar(255)
is_valid tinyint
create_date datetime
update_date datetime

# t_feed

字段名 数据类型 说明
id int 主键
content varchar(255)
fk_diner_id int 外键
praise_amount int
comment_amount int
fk_restaurant_id int 外键
create_date datetime
update_date datetime
is_valid tinyint

# t_follow

字段名 数据类型 说明
id int 主键
diner_id int
follow_diner_id int
is_valid tinyint
create_date datetime
update_date datetime

# t_dictionary

字段名 数据类型 说明
id int 主键
type varchar(20)
data varchar(50)
display_order int

# t_voucher

字段名 数据类型 说明
id int 主键
title varchar(255)
thumbnail varchar(255)
amount int
price decimal(10,2)
status int
expire_time datetime
redeem_restaurant_id int
stock int
stock_left int
description varchar(255)
clause varchar(255)
create_time datetime
update_time datetime
is_valid tinyint

# t_voucher_order

字段名 数据类型 说明
id int 主键

qrcode varchar(255)
payment tinyint
status tinyint

update_time datetime
is_valid tinyint

average_price varchar(20)
avg_lunch_price decimal(19,0)
introduction varchar(200)
status int
create_date datetime

thumbnail varchar(120)
open_hours varchar(500)
like_votes int
amenities varchar(100)
tags varchar(100)
open_date datetime
close_date datetime
create_time datetime
update_time datetime
