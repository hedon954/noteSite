# gRPC

gRPC (opens new window)

gRPC Github (opens new window)

gRPC 是一种现代开源高性能远程过程调用 (RPC) 框架，可以在任何环境中运行。它可以通过对负载平衡、跟踪、健康检查和身份验证的可插拔支持，有效地连接数据中心内和数据中心之间的服务。它还适用于分布式计算的 last mile，将设备、移动应用程序和浏览器连接到后端服务。

# 1. Download

需要先下载 protocol compiler：

go install google.golang.org/protobuf/cmd/protoc-gen-go@v1.26
go install google.golang.org/grpc/cmd/protoc-gen-go-grpc@v1.1

配置环境变量：

export PATH="$PATH:$(go env GOPATH)/bin"

# 2. HelloWorld

下载 gRPC 官网的示例代码：

git clone -b v1.41.0 https://github.com/grpc/grpc-go

进入示例代码的目录：

cd grpc-go/examples/helloworld

运行服务端代码：

go run greeter_server/main.go

再运行客户端代码，会收到来自服务端的信息：

$ go run greeter_client/main.go
Greeting: Hello world

# 3. BestPractice

# 3.1 定义 .proto 文件

syntax = "proto3";
package coolcar;
option go_package="gRPC/gen/go;trippb";

message Location {
    double latitude = 1;
    double longitude = 2;
}

// 数字才是真正的顺序
message Trip{
    string start = 1;
    Location start_pos = 5;
    repeated Location path_locations = 7;   // repeated 表示复数，会变成数组
    string end = 2;
    Location end_pos = 6;
    int64 duration_sec = 3;
    int64 fee_cent = 4;
}

message GetTripRequest {
    string id = 1;
}

message GetTripResponse{
    string id = 1;
    Trip trip = 2;
}

// gRPC 服务
service TripService {
		// 服务中的一个接口
    rpc GetTrip (GetTripRequest) returns (GetTripResponse);
}

# 3.2 创建生成代码目录

mkdir -p gen/go

# 3.3 生成代码

protoc -I=. --go_out=plugins=grpc,paths=source_relative:gen/go trip.proto

生成的结构体：

type Location struct {
	state         protoimpl.MessageState
	sizeCache     protoimpl.SizeCache
	unknownFields protoimpl.UnknownFields

	Latitude  float64 `protobuf:"fixed64,1,opt,name=latitude,proto3" json:"latitude,omitempty"`
	Longitude float64 `protobuf:"fixed64,2,opt,name=longitude,proto3" json:"longitude,omitempty"`
}

type Trip struct {
	state         protoimpl.MessageState
	sizeCache     protoimpl.SizeCache
	unknownFields protoimpl.UnknownFields

	Start         string      `protobuf:"bytes,1,opt,name=start,proto3" json:"start,omitempty"`
	StartPos      *Location   `protobuf:"bytes,5,opt,name=start_pos,json=startPos,proto3" json:"start_pos,omitempty"`
	PathLocations []*Location `protobuf:"bytes,7,rep,name=path_locations,json=pathLocations,proto3" json:"path_locations,omitempty"` // repeated 表示复数，会变成数组
	End           string      `protobuf:"bytes,2,opt,name=end,proto3" json:"end,omitempty"`
	EndPos        *Location   `protobuf:"bytes,6,opt,name=end_pos,json=endPos,proto3" json:"end_pos,omitempty"`
	DurationSec   int64       `protobuf:"varint,3,opt,name=duration_sec,json=durationSec,proto3" json:"duration_sec,omitempty"`
	FeeCent       int64       `protobuf:"varint,4,opt,name=fee_cent,json=feeCent,proto3" json:"fee_cent,omitempty"`
}

生成的服务端代码：（部分）

// TripServiceServer is the server API for TripService service.
type TripServiceServer interface {
	GetTrip(context.Context, *GetTripRequest) (*GetTripResponse, error)
}

生成的客户端代码：（部分）

// TripServiceClient is the client API for TripService service.
//
// For semantics around ctx use and closing/ending streaming RPCs, please refer to https://godoc.org/google.golang.org/grpc#ClientConn.NewStream.
type TripServiceClient interface {
	GetTrip(ctx context.Context, in *GetTripRequest, opts ...grpc.CallOption) (*GetTripResponse, error)
}

上面是 gRPC 为我们生成的接口，我们需要分别的服务端和客户端实现该接口。

# 3.4 实现服务端

生成服务端目录：

mkdir -p server/tripservice
cd server/tripservice

生成服务端文件：

touch main.go

实现 GetTrip 接口：

package trip

import (
	"context"
	trippb "gRPCStudy/gen/go"
)


type Service struct {
}

/**
 * GetTrip(context.Context, *GetTripRequest) (*GetTripResponse, error)
 */
func (*Service) GetTrip(c context.Context, in *trippb.GetTripRequest) (*trippb.GetTripResponse, error) {
	return &trippb.GetTripResponse{
		Id:   in.Id,
		Trip: &trippb.Trip{Start: "abc", End: "def", DurationSec: 3600, FeeCent: 2000},
	}, nil
}

启动一个 gRPC 服务：

# 回退到 server 目录
cd ..
# 创建 main.go
touch main.go

编写 main.go：

package main

import (
	"context"
	"log"
	"net"
	"net/http"
  
	trippb "gRPCStudy/gen/go"
	trip "gRPCStudy/server/tripservice"

	"google.golang.org/grpc"
)

func main() {
  
	log.SetFlags(log.Lshortfile)
	
  // 启动一个 tcp 监听器
	listener, err := net.Listen("tcp", ":8081")
	if err != nil {
		log.Fatalf("failed to listen: %v", err)
	}
  
  // 启动一个 gRPC 服务
	s := grpc.NewServer()
  
  // 调用 gRPC API 注册服务端
	trippb.RegisterTripServiceServer(s, &trip.Service{})
	log.Fatal(s.Serve(listener))
}

启动服务端：

# 回退到项目根目录
cd ..
# 运行服务端
go run server/main.go

# 3.5 实现客户端

创建客户端 main.go:

mkdir client
cd client
touch main.go

编写 main.go:

package main

import (
	"context"
	"fmt"
	"log"
	trippb "gRPCStudy/gen/go"

	"google.golang.org/grpc"
)

func main() {
  
  // 连接服务端
	conn, err := grpc.Dial("localhost:8081", grpc.WithInsecure())
	if err != nil {
		log.Fatalf("cannot connect server: %v", err)
	}
	
  // 创建一个 gRPC 客户端
	tsClient := trippb.NewTripServiceClient(conn)
  
  // 调用 gRPC 接口
	resp, err := tsClient.GetTrip(context.Background(), &trippb.GetTripRequest{
		Id: "10086",
	})

	if err != nil {
		log.Fatalf("get trip failed, err: %v", err)
	}
  
  // 输出调用结果
	fmt.Println(resp)
}

至此，一个 gRPC 的最佳实践就完成了！
